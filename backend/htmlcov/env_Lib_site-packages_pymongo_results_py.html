20class _WriteResult(object): 

21 """Base class for write result classes.""" 

22 

23 __slots__ = ("__acknowledged",) 

24 

25 def __init__(self, acknowledged): 

26 self.__acknowledged = acknowledged 

27 

28 def _raise_if_unacknowledged(self, property_name): 

29 """Raise an exception on property access if unacknowledged.""" 

30 if not self.__acknowledged: 

31 raise InvalidOperation("A value for %s is not available when " 

32 "the write is unacknowledged. Check the " 

33 "acknowledged attribute to avoid this " 

34 "error." % (property_name,)) 

35 

36 @property 

37 def acknowledged(self): 

38 """Is this the result of an acknowledged write operation? 

39 

40 The :attr:`acknowledged` attribute will be ``False`` when using 

41 ``WriteConcern(w=0)``, otherwise ``True``. 

42 

43 .. note:: 

44 If the :attr:`acknowledged` attribute is ``False`` all other 

45 attibutes of this class will raise 

46 :class:`~pymongo.errors.InvalidOperation` when accessed. Values for 

47 other attributes cannot be determined if the write operation was 

48 unacknowledged. 

49 

50 .. seealso:: 

51 :class:`~pymongo.write_concern.WriteConcern` 

52 """ 

53 return self.__acknowledged

111 @property

114 self._raise_if_unacknowledged("matched_count")
