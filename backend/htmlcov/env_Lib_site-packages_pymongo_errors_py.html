31class PyMongoError(Exception): 

32 """Base class for all PyMongo exceptions.""" 

33 def __init__(self, message='', error_labels=None): 

34 super(PyMongoError, self).__init__(message) 

35 self._message = message 

36 self._error_labels = set(error_labels or []) 

37 

38 def has_error_label(self, label): 

39 """Return True if this error contains the given label. 

40 

41 .. versionadded:: 3.7 

42 """ 

43 return label in self._error_labels 

44 

45 def _add_error_label(self, label): 

46 """Add the given label to this error.""" 

47 self._error_labels.add(label) 

48 

49 def _remove_error_label(self, label): 

50 """Remove the given label from this error.""" 

51 self._error_labels.discard(label)

73class AutoReconnect(ConnectionFailure): 

74 """Raised when a connection to the database is lost and an attempt to 

75 auto-reconnect will be made. 

76 

77 In order to auto-reconnect you must handle this exception, recognizing that 

78 the operation which caused it has not necessarily succeeded. Future 

79 operations will attempt to open a new connection to the database (and 

80 will continue to raise this exception until the first successful 

81 connection is made). 

82 

83 Subclass of :exc:`~pymongo.errors.ConnectionFailure`. 

84 """ 

85 def __init__(self, message='', errors=None): 

86 error_labels = None 

87 if errors is not None and isinstance(errors, dict): 

88 error_labels = errors.get('errorLabels') 

89 super(AutoReconnect, self).__init__(message, error_labels) 

90 self.errors = self.details = errors or []

111class NotMasterError(AutoReconnect): 

112 """The server responded "not master" or "node is recovering". 

113 

114 These errors result from a query, write, or command. The operation failed 

115 because the client thought it was using the primary but the primary has 

116 stepped down, or the client thought it was using a healthy secondary but 

117 the secondary is stale and trying to recover. 

118 

119 The client launches a refresh operation on a background thread, to update 

120 its view of the server as soon as possible after throwing this exception. 

121 

122 Subclass of :exc:`~pymongo.errors.AutoReconnect`.
