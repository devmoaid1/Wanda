29 """Options to configure automatic client-side field level encryption."""

31 def __init__(self, kms_providers, key_vault_namespace, 

32 key_vault_client=None, schema_map=None, 

33 bypass_auto_encryption=False, 

34 mongocryptd_uri='mongodb://localhost:27020', 

35 mongocryptd_bypass_spawn=False, 

36 mongocryptd_spawn_path='mongocryptd', 

37 mongocryptd_spawn_args=None): 

38 """Options to configure automatic client-side field level encryption. 

39 

40 Automatic client-side field level encryption requires MongoDB 4.2 

41 enterprise or a MongoDB 4.2 Atlas cluster. Automatic encryption is not 

42 supported for operations on a database or view and will result in 

43 error. 

44 

45 Although automatic encryption requires MongoDB 4.2 enterprise or a 

46 MongoDB 4.2 Atlas cluster, automatic *decryption* is supported for all 

47 users. To configure automatic *decryption* without automatic 

48 *encryption* set ``bypass_auto_encryption=True``. Explicit 

49 encryption and explicit decryption is also supported for all users 

50 with the :class:`~pymongo.encryption.ClientEncryption` class. 

51 

52 See :ref:`automatic-client-side-encryption` for an example. 

53 

54 :Parameters: 

55 - `kms_providers`: Map of KMS provider options. Two KMS providers 

56 are supported: "aws" and "local". The kmsProviders map values 

57 differ by provider: 

58 

59 - `aws`: Map with "accessKeyId" and "secretAccessKey" as strings. 

60 These are the AWS access key ID and AWS secret access key used 

61 to generate KMS messages. 

62 - `azure`: Map with "tenantId", "clientId", and "clientSecret" as 

63 strings. Additionally, "identityPlatformEndpoint" may also be 

64 specified as a string (defaults to 'login.microsoftonline.com'). 

65 These are the Azure Active Directory credentials used to 

66 generate Azure Key Vault messages. 

67 - `gcp`: Map with "email" as a string and "privateKey" 

68 as `bytes` or a base64 encoded string (unicode on Python 2). 

69 Additionally, "endpoint" may also be specified as a string 

70 (defaults to 'oauth2.googleapis.com'). These are the 

71 credentials used to generate Google Cloud KMS messages. 

72 - `local`: Map with "key" as `bytes` (96 bytes in length) or 

73 a base64 encoded string (unicode on Python 2) which decodes 

74 to 96 bytes. "key" is the master key used to encrypt/decrypt 

75 data keys. This key should be generated and stored as securely 

76 as possible. 

77 

78 - `key_vault_namespace`: The namespace for the key vault collection. 

79 The key vault collection contains all data keys used for encryption 

80 and decryption. Data keys are stored as documents in this MongoDB 

81 collection. Data keys are protected with encryption by a KMS 

82 provider. 

83 - `key_vault_client` (optional): By default the key vault collection 

84 is assumed to reside in the same MongoDB cluster as the encrypted 

85 MongoClient. Use this option to route data key queries to a 

86 separate MongoDB cluster. 

87 - `schema_map` (optional): Map of collection namespace ("db.coll") to 

88 JSON Schema. By default, a collection's JSONSchema is periodically 

89 polled with the listCollections command. But a JSONSchema may be 

90 specified locally with the schemaMap option. 

91 

92 **Supplying a `schema_map` provides more security than relying on 

93 JSON Schemas obtained from the server. It protects against a 

94 malicious server advertising a false JSON Schema, which could trick 

95 the client into sending unencrypted data that should be 

96 encrypted.** 

97 

98 Schemas supplied in the schemaMap only apply to configuring 

99 automatic encryption for client side encryption. Other validation 

100 rules in the JSON schema will not be enforced by the driver and 

101 will result in an error. 

102 - `bypass_auto_encryption` (optional): If ``True``, automatic 

103 encryption will be disabled but automatic decryption will still be 

104 enabled. Defaults to ``False``. 

105 - `mongocryptd_uri` (optional): The MongoDB URI used to connect 

106 to the *local* mongocryptd process. Defaults to 

107 ``'mongodb://localhost:27020'``. 

108 - `mongocryptd_bypass_spawn` (optional): If ``True``, the encrypted 

109 MongoClient will not attempt to spawn the mongocryptd process. 

110 Defaults to ``False``. 

111 - `mongocryptd_spawn_path` (optional): Used for spawning the 

112 mongocryptd process. Defaults to ``'mongocryptd'`` and spawns 

113 mongocryptd from the system path. 

114 - `mongocryptd_spawn_args` (optional): A list of string arguments to 

115 use when spawning the mongocryptd process. Defaults to 

116 ``['--idleShutdownTimeoutSecs=60']``. If the list does not include 

117 the ``idleShutdownTimeoutSecs`` option then 

118 ``'--idleShutdownTimeoutSecs=60'`` will be added. 

119 

120 .. versionadded:: 3.9 

121 """ 

122 if not _HAVE_PYMONGOCRYPT: 

123 raise ConfigurationError( 

124 "client side encryption requires the pymongocrypt library: " 

125 "install a compatible version with: " 

126 "python -m pip install 'pymongo[encryption]'") 

127 

128 self._kms_providers = kms_providers 

129 self._key_vault_namespace = key_vault_namespace 

130 self._key_vault_client = key_vault_client 

131 self._schema_map = schema_map 

132 self._bypass_auto_encryption = bypass_auto_encryption 

133 self._mongocryptd_uri = mongocryptd_uri 

134 self._mongocryptd_bypass_spawn = mongocryptd_bypass_spawn 

135 self._mongocryptd_spawn_path = mongocryptd_spawn_path 

136 self._mongocryptd_spawn_args = (copy.copy(mongocryptd_spawn_args) or 

137 ['--idleShutdownTimeoutSecs=60']) 

138 if not isinstance(self._mongocryptd_spawn_args, list): 

139 raise TypeError('mongocryptd_spawn_args must be a list') 

140 if not any('idleShutdownTimeoutSecs' in s 

141 for s in self._mongocryptd_spawn_args): 

142 self._mongocryptd_spawn_args.append('--idleShutdownTimeoutSecs=60')
