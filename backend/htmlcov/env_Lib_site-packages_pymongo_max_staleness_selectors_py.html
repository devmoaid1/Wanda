15"""Criteria to select ServerDescriptions based on maxStalenessSeconds. 

16 

17The Max Staleness Spec says: When there is a known primary P, 

18a secondary S's staleness is estimated with this formula: 

19 

20 (S.lastUpdateTime - S.lastWriteDate) - (P.lastUpdateTime - P.lastWriteDate) 

21 + heartbeatFrequencyMS 

22 

23When there is no known primary, a secondary S's staleness is estimated with: 

24 

25 SMax.lastWriteDate - S.lastWriteDate + heartbeatFrequencyMS 

26 

27where "SMax" is the secondary with the greatest lastWriteDate. 

28""" 

29 

30from pymongo.errors import ConfigurationError 

31from pymongo.server_type import SERVER_TYPE 

32 

33 

34# Constant defined in Max Staleness Spec: An idle primary writes a no-op every 

35# 10 seconds to refresh secondaries' lastWriteDate values. 

36IDLE_WRITE_PERIOD = 10 

37SMALLEST_MAX_STALENESS = 90 

38 

39 

40def _validate_max_staleness(max_staleness, 

41 heartbeat_frequency): 

42 # We checked for max staleness -1 before this, it must be positive here. 

43 if max_staleness < heartbeat_frequency + IDLE_WRITE_PERIOD: 

44 raise ConfigurationError( 

45 "maxStalenessSeconds must be at least heartbeatFrequencyMS +" 

46 " %d seconds. maxStalenessSeconds is set to %d," 

47 " heartbeatFrequencyMS is set to %d." % ( 

48 IDLE_WRITE_PERIOD, max_staleness, heartbeat_frequency * 1000)) 

49 

50 if max_staleness < SMALLEST_MAX_STALENESS: 

51 raise ConfigurationError( 

52 "maxStalenessSeconds must be at least %d. " 

53 "maxStalenessSeconds is set to %d." % ( 

54 SMALLEST_MAX_STALENESS, max_staleness)) 

55 

56 

57def _with_primary(max_staleness, selection): 

58 """Apply max_staleness, in seconds, to a Selection with a known primary.""" 

59 primary = selection.primary 

60 sds = [] 

61 

62 for s in selection.server_descriptions: 

63 if s.server_type == SERVER_TYPE.RSSecondary: 

64 # See max-staleness.rst for explanation of this formula. 

65 staleness = ( 

66 (s.last_update_time - s.last_write_date) - 

67 (primary.last_update_time - primary.last_write_date) + 

68 selection.heartbeat_frequency) 

69 

70 if staleness <= max_staleness: 

71 sds.append(s) 

72 else: 

73 sds.append(s) 

74 

75 return selection.with_server_descriptions(sds) 

76 

77 

78def _no_primary(max_staleness, selection): 

79 """Apply max_staleness, in seconds, to a Selection with no known primary.""" 

80 # Secondary that's replicated the most recent writes. 

81 smax = selection.secondary_with_max_last_write_date() 

82 if not smax: 

83 # No secondaries and no primary, short-circuit out of here. 

84 return selection.with_server_descriptions([]) 

85 

86 sds = [] 

87 

88 for s in selection.server_descriptions: 

89 if s.server_type == SERVER_TYPE.RSSecondary: 

90 # See max-staleness.rst for explanation of this formula. 

91 staleness = (smax.last_write_date - 

92 s.last_write_date + 

93 selection.heartbeat_frequency) 

94 

95 if staleness <= max_staleness: 

96 sds.append(s) 

97 else: 

98 sds.append(s) 

99 

100 return selection.with_server_descriptions(sds)

103def select(max_staleness, selection): 

104 """Apply max_staleness, in seconds, to a Selection.""" 

105 if max_staleness == -1: 

106 return selection 

107 

108 # Server Selection Spec: If the TopologyType is ReplicaSetWithPrimary or 

109 # ReplicaSetNoPrimary, a client MUST raise an error if maxStaleness < 

110 # heartbeatFrequency + IDLE_WRITE_PERIOD, or if maxStaleness < 90. 

111 _validate_max_staleness(max_staleness, selection.heartbeat_frequency) 

112 

113 if selection.primary: 

114 return _with_primary(max_staleness, selection) 

115 else: 

116 return _no_primary(max_staleness, selection)
