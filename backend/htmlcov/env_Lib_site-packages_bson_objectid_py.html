63 def __init__(self, oid=None): 

64 """Initialize a new ObjectId. 

65 

66 An ObjectId is a 12-byte unique identifier consisting of: 

67 

68 - a 4-byte value representing the seconds since the Unix epoch, 

69 - a 5-byte random value, 

70 - a 3-byte counter, starting with a random value. 

71 

72 By default, ``ObjectId()`` creates a new unique identifier. The 

73 optional parameter `oid` can be an :class:`ObjectId`, or any 12 

74 :class:`bytes` or, in Python 2, any 12-character :class:`str`. 

75 

76 For example, the 12 bytes b'foo-bar-quux' do not follow the ObjectId 

77 specification but they are acceptable input:: 

78 

79 >>> ObjectId(b'foo-bar-quux') 

80 ObjectId('666f6f2d6261722d71757578') 

81 

82 `oid` can also be a :class:`unicode` or :class:`str` of 24 hex digits:: 

83 

84 >>> ObjectId('0123456789ab0123456789ab') 

85 ObjectId('0123456789ab0123456789ab') 

86 >>> 

87 >>> # A u-prefixed unicode literal: 

88 >>> ObjectId(u'0123456789ab0123456789ab') 

89 ObjectId('0123456789ab0123456789ab') 

90 

91 Raises :class:`~bson.errors.InvalidId` if `oid` is not 12 bytes nor 

92 24 hex digits, or :class:`TypeError` if `oid` is not an accepted type. 

93 

94 :Parameters: 

95 - `oid` (optional): a valid ObjectId. 

96 

97 .. mongodoc:: objectids 

98 

99 .. versionchanged:: 3.8 

100 :class:`~bson.objectid.ObjectId` now implements the `ObjectID 

101 specification version 0.2 

102 <https://github.com/mongodb/specifications/blob/master/source/ 

103 objectid.rst>`_. 

104 """ 

105 if oid is None: 

106 self.__generate() 

107 elif isinstance(oid, bytes) and len(oid) == 12: 

108 self.__id = oid 

109 else: 

110 self.__validate(oid)

112 @classmethod 

113 def from_datetime(cls, generation_time): 

114 """Create a dummy ObjectId instance with a specific generation time. 

115 

116 This method is useful for doing range queries on a field 

117 containing :class:`ObjectId` instances. 

118 

119 .. warning:: 

120 It is not safe to insert a document containing an ObjectId 

121 generated using this method. This method deliberately 

122 eliminates the uniqueness guarantee that ObjectIds 

123 generally provide. ObjectIds generated with this method 

124 should be used exclusively in queries. 

125 

126 `generation_time` will be converted to UTC. Naive datetime 

127 instances will be treated as though they already contain UTC. 

128 

129 An example using this helper to get documents where ``"_id"`` 

130 was generated before January 1, 2010 would be: 

131 

132 >>> gen_time = datetime.datetime(2010, 1, 1) 

133 >>> dummy_id = ObjectId.from_datetime(gen_time) 

134 >>> result = collection.find({"_id": {"$lt": dummy_id}}) 

135 

136 :Parameters: 

137 - `generation_time`: :class:`~datetime.datetime` to be used 

138 as the generation time for the resulting ObjectId. 

139 """ 

140 if generation_time.utcoffset() is not None: 

141 generation_time = generation_time - generation_time.utcoffset() 

142 timestamp = calendar.timegm(generation_time.timetuple()) 

143 oid = struct.pack( 

144 ">I", int(timestamp)) + b"\x00\x00\x00\x00\x00\x00\x00\x00" 

145 return cls(oid)

175 def __generate(self): 

176 """Generate a new value for this ObjectId. 

177 """ 

178 

179 # 4 bytes current time 

180 oid = struct.pack(">I", int(time.time())) 

181 

182 # 5 bytes random 

183 oid += ObjectId._random() 

184 

185 # 3 bytes inc 

186 with ObjectId._inc_lock: 

187 oid += struct.pack(">I", ObjectId._inc)[1:4] 

188 ObjectId._inc = (ObjectId._inc + 1) % (_MAX_COUNTER_VALUE + 1) 

189 

190 self.__id = oid

192 def __validate(self, oid): 

193 """Validate and use the given id for this ObjectId. 

194 

195 Raises TypeError if id is not an instance of 

196 (:class:`basestring` (:class:`str` or :class:`bytes` 

197 in python 3), ObjectId) and InvalidId if it is not a 

198 valid ObjectId. 

199 

200 :Parameters: 

201 - `oid`: a valid ObjectId 

202 """ 

203 if isinstance(oid, ObjectId): 

204 self.__id = oid.binary 

205 # bytes or unicode in python 2, str in python 3 

206 elif isinstance(oid, string_type): 

207 if len(oid) == 24: 

208 try: 

209 self.__id = bytes_from_hex(oid) 

210 except (TypeError, ValueError): 

211 _raise_invalid_id(oid) 

212 else: 

213 _raise_invalid_id(oid) 

214 else: 

215 raise TypeError("id must be an instance of (bytes, %s, ObjectId), " 

216 "not %s" % (text_type.__name__, type(oid)))
