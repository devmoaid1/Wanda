15"""Tools for representing raw BSON documents. 

16""" 

17 

18from bson import _raw_to_dict, _get_object_size 

19from bson.py3compat import abc, iteritems 

20from bson.codec_options import ( 

21 DEFAULT_CODEC_OPTIONS as DEFAULT, _RAW_BSON_DOCUMENT_MARKER) 

22from bson.son import SON 

23 

24 

25class RawBSONDocument(abc.Mapping): 

26 """Representation for a MongoDB document that provides access to the raw 

27 BSON bytes that compose it. 

28 

29 Only when a field is accessed or modified within the document does 

30 RawBSONDocument decode its bytes. 

31 """ 

32 

33 __slots__ = ('__raw', '__inflated_doc', '__codec_options') 

34 _type_marker = _RAW_BSON_DOCUMENT_MARKER 

35 

36 def __init__(self, bson_bytes, codec_options=None): 

37 """Create a new :class:`RawBSONDocument` 

38 

39 :class:`RawBSONDocument` is a representation of a BSON document that 

40 provides access to the underlying raw BSON bytes. Only when a field is 

41 accessed or modified within the document does RawBSONDocument decode 

42 its bytes. 

43 

44 :class:`RawBSONDocument` implements the ``Mapping`` abstract base 

45 class from the standard library so it can be used like a read-only 

46 ``dict``:: 

47 

48 >>> from bson import encode 

49 >>> raw_doc = RawBSONDocument(encode({'_id': 'my_doc'})) 

50 >>> raw_doc.raw 

51 b'...' 

52 >>> raw_doc['_id'] 

53 'my_doc' 

54 

55 :Parameters: 

56 - `bson_bytes`: the BSON bytes that compose this document 

57 - `codec_options` (optional): An instance of 

58 :class:`~bson.codec_options.CodecOptions` whose ``document_class`` 

59 must be :class:`RawBSONDocument`. The default is 

60 :attr:`DEFAULT_RAW_BSON_OPTIONS`. 

61 

62 .. versionchanged:: 3.8 

63 :class:`RawBSONDocument` now validates that the ``bson_bytes`` 

64 passed in represent a single bson document. 

65 

66 .. versionchanged:: 3.5 

67 If a :class:`~bson.codec_options.CodecOptions` is passed in, its 

68 `document_class` must be :class:`RawBSONDocument`. 

69 """ 

70 self.__raw = bson_bytes 

71 self.__inflated_doc = None 

72 # Can't default codec_options to DEFAULT_RAW_BSON_OPTIONS in signature, 

73 # it refers to this class RawBSONDocument. 

74 if codec_options is None: 

75 codec_options = DEFAULT_RAW_BSON_OPTIONS 

76 elif codec_options.document_class is not RawBSONDocument: 

77 raise TypeError( 

78 "RawBSONDocument cannot use CodecOptions with document " 

79 "class %s" % (codec_options.document_class, )) 

80 self.__codec_options = codec_options 

81 # Validate the bson object size. 

82 _get_object_size(bson_bytes, 0, len(bson_bytes)) 

83 

84 @property 

85 def raw(self): 

86 """The raw BSON bytes composing this document.""" 

87 return self.__raw 

88 

89 def items(self): 

90 """Lazily decode and iterate elements in this document.""" 

91 return iteritems(self.__inflated) 

92 

93 @property 

94 def __inflated(self): 

95 if self.__inflated_doc is None: 

96 # We already validated the object's size when this document was 

97 # created, so no need to do that again. 

98 # Use SON to preserve ordering of elements. 

99 self.__inflated_doc = _inflate_bson( 

100 self.__raw, self.__codec_options) 

101 return self.__inflated_doc 

102 

103 def __getitem__(self, item): 

104 return self.__inflated[item] 

105 

106 def __iter__(self): 

107 return iter(self.__inflated) 

108 

109 def __len__(self): 

110 return len(self.__inflated) 

111 

112 def __eq__(self, other): 

113 if isinstance(other, RawBSONDocument): 

114 return self.__raw == other.raw 

115 return NotImplemented 

116 

117 def __repr__(self): 

118 return ("RawBSONDocument(%r, codec_options=%r)" 

119 % (self.raw, self.__codec_options)) 

120 

121 

122def _inflate_bson(bson_bytes, codec_options): 

123 """Inflates the top level fields of a BSON document. 

124 

125 :Parameters: 

126 - `bson_bytes`: the BSON bytes that compose this document 

127 - `codec_options`: An instance of 

128 :class:`~bson.codec_options.CodecOptions` whose ``document_class`` 

129 must be :class:`RawBSONDocument`. 

130 """ 

131 # Use SON to preserve ordering of elements. 

132 return _raw_to_dict( 

133 bson_bytes, 4, len(bson_bytes)-1, codec_options, SON()) 

134 

135 

136DEFAULT_RAW_BSON_OPTIONS = DEFAULT.with_options(document_class=RawBSONDocument) 

137"""The default :class:`~bson.codec_options.CodecOptions` for 

138:class:`RawBSONDocument`. 

139"""
