33_SHUTDOWN_CODES = frozenset([ 

34 11600, # InterruptedAtShutdown

58def _gen_index_name(keys): 

59 """Generate an index name from the set of fields it is over.""" 

60 return _UUNDER.join(["%s_%s" % item for item in keys])

63def _index_list(key_or_list, direction=None): 

64 """Helper to generate a list of (key, direction) pairs. 

65 

66 Takes such a list, or a single key, or a single key and direction. 

67 """ 

68 if direction is not None: 

69 return [(key_or_list, direction)] 

70 else: 

71 if isinstance(key_or_list, string_type): 

72 return [(key_or_list, ASCENDING)] 

73 elif not isinstance(key_or_list, (list, tuple)): 

74 raise TypeError("if no direction is specified, " 

75 "key_or_list must be an instance of list") 

76 return key_or_list

79def _index_document(index_list): 

80 """Helper to generate an index specifying document. 

81 

82 Takes a list of (key, direction) pairs. 

83 """ 

84 if isinstance(index_list, abc.Mapping): 

85 raise TypeError("passing a dict to sort/create_index/hint is not " 

86 "allowed - use a list of tuples instead. did you " 

87 "mean %r?" % list(iteritems(index_list))) 

88 elif not isinstance(index_list, (list, tuple)): 

89 raise TypeError("must use a list of (key, direction) pairs, " 

90 "not: " + repr(index_list)) 

91 if not len(index_list): 

92 raise ValueError("key_or_list must not be the empty list") 

93 

94 index = SON() 

95 for (key, value) in index_list: 

96 if not isinstance(key, string_type): 

97 raise TypeError("first item in each key pair must be a string") 

98 if not isinstance(value, (string_type, int, abc.Mapping)): 

99 raise TypeError("second item in each key pair must be 1, -1, " 

100 "'2d', or another valid MongoDB index specifier.") 

101 index[key] = value 

102 return index

105def _check_command_response(response, max_wire_version, 

106 allowable_errors=None, 

107 parse_write_concern_error=False): 

108 """Check the response to a command for errors. 

109 """ 

110 if "ok" not in response: 

111 # Server didn't recognize our message as a command. 

112 raise OperationFailure(response.get("$err"), 

113 response.get("code"), 

114 response, 

115 max_wire_version) 

116 

117 if parse_write_concern_error and 'writeConcernError' in response: 

118 _error = response["writeConcernError"] 

119 _labels = response.get("errorLabels") 

120 if _labels: 

121 _error.update({'errorLabels': _labels}) 

122 _raise_write_concern_error(_error) 

123 

124 if response["ok"]: 

125 return 

126 

127 details = response 

128 # Mongos returns the error details in a 'raw' object 

129 # for some errors. 

130 if "raw" in response: 

131 for shard in itervalues(response["raw"]): 

132 # Grab the first non-empty raw error from a shard. 

133 if shard.get("errmsg") and not shard.get("ok"): 

134 details = shard 

135 break 

136 

137 errmsg = details["errmsg"] 

138 code = details.get("code") 

139 

140 # For allowable errors, only check for error messages when the code is not 

141 # included. 

142 if allowable_errors: 

143 if code is not None: 

144 if code in allowable_errors: 

145 return 

146 elif errmsg in allowable_errors: 

147 return 

148 

149 # Server is "not master" or "recovering" 

150 if code in _NOT_MASTER_CODES: 

151 raise NotMasterError(errmsg, response) 

152 elif "not master" in errmsg or "node is recovering" in errmsg: 

153 raise NotMasterError(errmsg, response) 

154 

155 # Other errors 

156 # findAndModify with upsert can raise duplicate key error 

157 if code in (11000, 11001, 12582): 

158 raise DuplicateKeyError(errmsg, code, response, max_wire_version) 

159 elif code == 50: 

160 raise ExecutionTimeout(errmsg, code, response, max_wire_version) 

161 elif code == 43: 

162 raise CursorNotFound(errmsg, code, response, max_wire_version) 

163 

164 raise OperationFailure(errmsg, code, response, max_wire_version)

167def _check_gle_response(result, max_wire_version): 

168 """Return getlasterror response as a dict, or raise OperationFailure.""" 

169 # Did getlasterror itself fail? 

170 _check_command_response(result, max_wire_version) 

171 

172 if result.get("wtimeout", False): 

173 # MongoDB versions before 1.8.0 return the error message in an "errmsg" 

174 # field. If "errmsg" exists "err" will also exist set to None, so we 

175 # have to check for "errmsg" first. 

176 raise WTimeoutError(result.get("errmsg", result.get("err")), 

177 result.get("code"), 

178 result) 

179 

180 error_msg = result.get("err", "") 

181 if error_msg is None: 

182 return result 

183 

184 if error_msg.startswith("not master"): 

185 raise NotMasterError(error_msg, result) 

186 

187 details = result 

188 

189 # mongos returns the error code in an error object for some errors. 

190 if "errObjects" in result: 

191 for errobj in result["errObjects"]: 

192 if errobj.get("err") == error_msg: 

193 details = errobj 

194 break 

195 

196 code = details.get("code") 

197 if code in (11000, 11001, 12582): 

198 raise DuplicateKeyError(details["err"], code, result) 

199 raise OperationFailure(details["err"], code, result)

202def _raise_last_write_error(write_errors): 

203 # If the last batch had multiple errors only report 

204 # the last error to emulate continue_on_error. 

205 error = write_errors[-1] 

206 if error.get("code") == 11000: 

207 raise DuplicateKeyError(error.get("errmsg"), 11000, error) 

208 raise WriteError(error.get("errmsg"), error.get("code"), error)

212 if "errInfo" in error and error["errInfo"].get('wtimeout'):

247def _fields_list_to_dict(fields, option_name): 

248 """Takes a sequence of field names and returns a matching dictionary. 

249 

250 ["a", "b"] becomes {"a": 1, "b": 1} 

251 

252 and 

253 

254 ["a.b.c", "d", "a.c"] becomes {"a.b.c": 1, "d": 1, "a.c": 1} 

255 """ 

256 if isinstance(fields, abc.Mapping): 

257 return fields 

258 

259 if isinstance(fields, (abc.Sequence, abc.Set)): 

260 if not all(isinstance(field, string_type) for field in fields): 

261 raise TypeError("%s must be a list of key names, each an " 

262 "instance of %s" % (option_name, 

263 string_type.__name__)) 

264 return dict.fromkeys(fields, 1) 

265 

266 raise TypeError("%s must be a mapping or " 

267 "list of key names" % (option_name,))
