43def command(sock_info, dbname, spec, slave_ok, is_mongos, 

44 read_preference, codec_options, session, client, check=True, 

45 allowable_errors=None, address=None, 

46 check_keys=False, listeners=None, max_bson_size=None, 

47 read_concern=None, 

48 parse_write_concern_error=False, 

49 collation=None, 

50 compression_ctx=None, 

51 use_op_msg=False, 

52 unacknowledged=False, 

53 user_fields=None, 

54 exhaust_allowed=False): 

55 """Execute a command over the socket, or raise socket.error. 

56 

57 :Parameters: 

58 - `sock`: a raw socket instance 

59 - `dbname`: name of the database on which to run the command 

60 - `spec`: a command document as an ordered dict type, eg SON. 

61 - `slave_ok`: whether to set the SlaveOkay wire protocol bit 

62 - `is_mongos`: are we connected to a mongos? 

63 - `read_preference`: a read preference 

64 - `codec_options`: a CodecOptions instance 

65 - `session`: optional ClientSession instance. 

66 - `client`: optional MongoClient instance for updating $clusterTime. 

67 - `check`: raise OperationFailure if there are errors 

68 - `allowable_errors`: errors to ignore if `check` is True 

69 - `address`: the (host, port) of `sock` 

70 - `check_keys`: if True, check `spec` for invalid keys 

71 - `listeners`: An instance of :class:`~pymongo.monitoring.EventListeners` 

72 - `max_bson_size`: The maximum encoded bson size for this server 

73 - `read_concern`: The read concern for this command. 

74 - `parse_write_concern_error`: Whether to parse the ``writeConcernError`` 

75 field in the command response. 

76 - `collation`: The collation for this command. 

77 - `compression_ctx`: optional compression Context. 

78 - `use_op_msg`: True if we should use OP_MSG. 

79 - `unacknowledged`: True if this is an unacknowledged command. 

80 - `user_fields` (optional): Response fields that should be decoded 

81 using the TypeDecoders from codec_options, passed to 

82 bson._decode_all_selective. 

83 - `exhaust_allowed`: True if we should enable OP_MSG exhaustAllowed. 

84 """ 

85 name = next(iter(spec)) 

86 ns = dbname + '.$cmd' 

87 flags = 4 if slave_ok else 0 

88 

89 # Publish the original command document, perhaps with lsid and $clusterTime. 

90 orig = spec 

91 if is_mongos and not use_op_msg: 

92 spec = message._maybe_add_read_preference(spec, read_preference) 

93 if read_concern and not (session and session.in_transaction): 

94 if read_concern.level: 

95 spec['readConcern'] = read_concern.document 

96 if (session and session.options.causal_consistency 

97 and session.operation_time is not None): 

98 spec.setdefault( 

99 'readConcern', {})['afterClusterTime'] = session.operation_time 

100 if collation is not None: 

101 spec['collation'] = collation 

102 

103 publish = listeners is not None and listeners.enabled_for_commands 

104 if publish: 

105 start = datetime.datetime.now() 

106 

107 if compression_ctx and name.lower() in _NO_COMPRESSION: 

108 compression_ctx = None 

109 

110 if (client and client._encrypter and 

111 not client._encrypter._bypass_auto_encryption): 

112 spec = orig = client._encrypter.encrypt( 

113 dbname, spec, check_keys, codec_options) 

114 # We already checked the keys, no need to do it again. 

115 check_keys = False 

116 

117 if use_op_msg: 

118 flags = _OpMsg.MORE_TO_COME if unacknowledged else 0 

119 flags |= _OpMsg.EXHAUST_ALLOWED if exhaust_allowed else 0 

120 request_id, msg, size, max_doc_size = message._op_msg( 

121 flags, spec, dbname, read_preference, slave_ok, check_keys, 

122 codec_options, ctx=compression_ctx) 

123 # If this is an unacknowledged write then make sure the encoded doc(s) 

124 # are small enough, otherwise rely on the server to return an error. 

125 if (unacknowledged and max_bson_size is not None and 

126 max_doc_size > max_bson_size): 

127 message._raise_document_too_large(name, size, max_bson_size) 

128 else: 

129 request_id, msg, size = message.query( 

130 flags, ns, 0, -1, spec, None, codec_options, check_keys, 

131 compression_ctx) 

132 

133 if (max_bson_size is not None 

134 and size > max_bson_size + message._COMMAND_OVERHEAD): 

135 message._raise_document_too_large( 

136 name, size, max_bson_size + message._COMMAND_OVERHEAD) 

137 

138 if publish: 

139 encoding_duration = datetime.datetime.now() - start 

140 listeners.publish_command_start(orig, dbname, request_id, address) 

141 start = datetime.datetime.now() 

142 

143 try: 

144 sock_info.sock.sendall(msg) 

145 if use_op_msg and unacknowledged: 

146 # Unacknowledged, fake a successful command response. 

147 reply = None 

148 response_doc = {"ok": 1} 

149 else: 

150 reply = receive_message(sock_info, request_id) 

151 sock_info.more_to_come = reply.more_to_come 

152 unpacked_docs = reply.unpack_response( 

153 codec_options=codec_options, user_fields=user_fields) 

154 

155 response_doc = unpacked_docs[0] 

156 if client: 

157 client._process_response(response_doc, session) 

158 if check: 

159 helpers._check_command_response( 

160 response_doc, sock_info.max_wire_version, allowable_errors, 

161 parse_write_concern_error=parse_write_concern_error) 

162 except Exception as exc: 

163 if publish: 

164 duration = (datetime.datetime.now() - start) + encoding_duration 

165 if isinstance(exc, (NotMasterError, OperationFailure)): 

166 failure = exc.details 

167 else: 

168 failure = message._convert_exception(exc) 

169 listeners.publish_command_failure( 

170 duration, failure, name, request_id, address) 

171 raise 

172 if publish: 

173 duration = (datetime.datetime.now() - start) + encoding_duration 

174 listeners.publish_command_success( 

175 duration, response_doc, name, request_id, address) 

176 

177 if client and client._encrypter and reply: 

178 decrypted = client._encrypter.decrypt(reply.raw_command_response()) 

179 response_doc = _decode_all_selective(decrypted, codec_options, 

180 user_fields)[0] 

181 

182 return response_doc

186def receive_message(sock_info, request_id, max_message_size=MAX_MESSAGE_SIZE): 

187 """Receive a raw BSON message or raise socket.error.""" 

188 timeout = sock_info.sock.gettimeout() 

189 if timeout: 

190 deadline = time() + timeout 

191 else: 

192 deadline = None 

193 # Ignore the response's request id. 

194 length, _, response_to, op_code = _UNPACK_HEADER( 

195 _receive_data_on_socket(sock_info, 16, deadline)) 

196 # No request_id for exhaust cursor "getMore". 

197 if request_id is not None: 

198 if request_id != response_to: 

199 raise ProtocolError("Got response id %r but expected " 

200 "%r" % (response_to, request_id)) 

201 if length <= 16: 

202 raise ProtocolError("Message length (%r) not longer than standard " 

203 "message header size (16)" % (length,)) 

204 if length > max_message_size: 

205 raise ProtocolError("Message length (%r) is larger than server max " 

206 "message size (%r)" % (length, max_message_size)) 

207 if op_code == 2012: 

208 op_code, _, compressor_id = _UNPACK_COMPRESSION_HEADER( 

209 _receive_data_on_socket(sock_info, 9, deadline)) 

210 data = decompress( 

211 _receive_data_on_socket(sock_info, length - 25, deadline), 

212 compressor_id) 

213 else: 

214 data = _receive_data_on_socket(sock_info, length - 16, deadline) 

215 

216 try: 

217 unpack_reply = _UNPACK_REPLY[op_code] 

218 except KeyError: 

219 raise ProtocolError("Got opcode %r but expected " 

220 "%r" % (op_code, _UNPACK_REPLY.keys())) 

221 return unpack_reply(data)

227def wait_for_read(sock_info, deadline): 

228 """Block until at least one byte is read, or a timeout, or a cancel.""" 

229 context = sock_info.cancel_context 

230 # Only Monitor connections can be cancelled. 

231 if context: 

232 sock = sock_info.sock 

233 while True: 

234 # SSLSocket can have buffered data which won't be caught by select. 

235 if hasattr(sock, 'pending') and sock.pending() > 0: 

236 readable = True 

237 else: 

238 # Wait up to 500ms for the socket to become readable and then 

239 # check for cancellation. 

240 if deadline: 

241 timeout = max(min(deadline - time(), _POLL_TIMEOUT), 0.001) 

242 else: 

243 timeout = _POLL_TIMEOUT 

244 readable = sock_info.socket_checker.select( 

245 sock, read=True, timeout=timeout) 

246 if context.cancelled: 

247 raise _OperationCancelled('isMaster cancelled') 

248 if readable: 

249 return 

250 if deadline and time() > deadline: 

251 raise socket.timeout("timed out")

253# memoryview was introduced in Python 2.7 but we only use it on Python 3 

254# because before 2.7.4 the struct module did not support memoryview: 

255# https://bugs.python.org/issue10212. 

256# In Jython, using slice assignment on a memoryview results in a 

257# NullPointerException. 

258if not PY3: 

259 def _receive_data_on_socket(sock_info, length, deadline): 

260 buf = bytearray(length) 

261 i = 0 

262 while length: 

263 try: 

264 wait_for_read(sock_info, deadline) 

265 chunk = sock_info.sock.recv(length) 

266 except (IOError, OSError) as exc: 

267 if _errno_from_exception(exc) == errno.EINTR: 

268 continue 

269 raise 

270 if chunk == b"": 

271 raise AutoReconnect("connection closed") 

272 

273 buf[i:i + len(chunk)] = chunk 

274 i += len(chunk) 

275 length -= len(chunk) 

276 

277 return bytes(buf) 

278else: 

279 def _receive_data_on_socket(sock_info, length, deadline): 

280 buf = bytearray(length) 

281 mv = memoryview(buf) 

282 bytes_read = 0 

283 while bytes_read < length: 

284 try: 

285 wait_for_read(sock_info, deadline) 

286 chunk_length = sock_info.sock.recv_into(mv[bytes_read:]) 

287 except (IOError, OSError) as exc: 

288 if _errno_from_exception(exc) == errno.EINTR: 

289 continue 

290 raise 

291 if chunk_length == 0: 

292 raise AutoReconnect("connection closed") 

293 

294 bytes_read += chunk_length 

295 

296 return mv
