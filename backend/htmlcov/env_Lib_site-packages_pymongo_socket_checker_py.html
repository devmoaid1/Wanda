15"""Select / poll helper""" 

16 

17import errno 

18import select 

19import sys 

20 

21# PYTHON-2320: Jython does not fully support poll on SSL sockets, 

22# https://bugs.jython.org/issue2900 

23_HAVE_POLL = hasattr(select, "poll") and not sys.platform.startswith('java') 

24_SelectError = getattr(select, "error", OSError) 

25 

26 

27def _errno_from_exception(exc): 

28 if hasattr(exc, 'errno'): 

29 return exc.errno 

30 if exc.args: 

31 return exc.args[0] 

32 return None 

33 

34 

35class SocketChecker(object): 

36 

37 def __init__(self): 

38 if _HAVE_POLL: 

39 self._poller = select.poll() 

40 else: 

41 self._poller = None 

42 

43 def select(self, sock, read=False, write=False, timeout=0): 

44 """Select for reads or writes with a timeout in seconds (or None). 

45 

46 Returns True if the socket is readable/writable, False on timeout. 

47 """ 

48 while True: 

49 try: 

50 if self._poller: 

51 mask = select.POLLERR | select.POLLHUP 

52 if read: 

53 mask = mask | select.POLLIN | select.POLLPRI 

54 if write: 

55 mask = mask | select.POLLOUT 

56 self._poller.register(sock, mask) 

57 try: 

58 # poll() timeout is in milliseconds. select() 

59 # timeout is in seconds. 

60 timeout_ = None if timeout is None else timeout * 1000 

61 res = self._poller.poll(timeout_) 

62 # poll returns a possibly-empty list containing 

63 # (fd, event) 2-tuples for the descriptors that have 

64 # events or errors to report. Return True if the list 

65 # is not empty. 

66 return bool(res) 

67 finally: 

68 self._poller.unregister(sock) 

69 else: 

70 rlist = [sock] if read else [] 

71 wlist = [sock] if write else [] 

72 res = select.select(rlist, wlist, [sock], timeout) 

73 # select returns a 3-tuple of lists of objects that are 

74 # ready: subsets of the first three arguments. Return 

75 # True if any of the lists are not empty. 

76 return any(res) 

77 except (_SelectError, IOError) as exc: 

78 if _errno_from_exception(exc) in (errno.EINTR, errno.EAGAIN): 

79 continue 

80 raise 

81 

82 def socket_closed(self, sock): 

83 """Return True if we know socket has been closed, False otherwise. 

84 """ 

85 try: 

86 return self.select(sock, read=True) 

87 except (RuntimeError, KeyError): 

88 # RuntimeError is raised during a concurrent poll. KeyError 

89 # is raised by unregister if the socket is not in the poller. 

90 # These errors should not be possible since we protect the 

91 # poller with a mutex. 

92 raise 

93 except ValueError: 

94 # ValueError is raised by register/unregister/select if the 

95 # socket file descriptor is negative or outside the range for 

96 # select (> 1023). 

97 return True 

98 except Exception: 

99 # Any other exceptions should be attributed to a closed 

100 # or invalid socket. 

101 return True
