27class Timestamp(object): 

28 """MongoDB internal timestamps used in the opLog. 

29 """ 

30 

31 _type_marker = 17 

32 

33 def __init__(self, time, inc): 

34 """Create a new :class:`Timestamp`. 

35 

36 This class is only for use with the MongoDB opLog. If you need 

37 to store a regular timestamp, please use a 

38 :class:`~datetime.datetime`. 

39 

40 Raises :class:`TypeError` if `time` is not an instance of 

41 :class: `int` or :class:`~datetime.datetime`, or `inc` is not 

42 an instance of :class:`int`. Raises :class:`ValueError` if 

43 `time` or `inc` is not in [0, 2**32). 

44 

45 :Parameters: 

46 - `time`: time in seconds since epoch UTC, or a naive UTC 

47 :class:`~datetime.datetime`, or an aware 

48 :class:`~datetime.datetime` 

49 - `inc`: the incrementing counter 

50 """ 

51 if isinstance(time, datetime.datetime): 

52 if time.utcoffset() is not None: 

53 time = time - time.utcoffset() 

54 time = int(calendar.timegm(time.timetuple())) 

55 if not isinstance(time, integer_types): 

56 raise TypeError("time must be an instance of int") 

57 if not isinstance(inc, integer_types): 

58 raise TypeError("inc must be an instance of int") 

59 if not 0 <= time < UPPERBOUND: 

60 raise ValueError("time must be contained in [0, 2**32)") 

61 if not 0 <= inc < UPPERBOUND: 

62 raise ValueError("inc must be contained in [0, 2**32)") 

63 

64 self.__time = time 

65 self.__inc = inc 

66

71 return self.__time
